            function saveData() {

                const data = {

                    budget: document.getElementById('budget').value,

                    cards: []

                };

                for (let i = 0; i < cardCount; i++) {

                    const cardData = {

                        level: document.getElementById(`level-${i}`).value,

                        awaken: document.getElementById(`awaken-${i}`).value,

                        stats: []

                    };

                    for (let j = 0; j < 3; j++) {

                        cardData.stats.push({

                            type: document.getElementById(`stat-type-${i}-${j}`).value,

                            value: document.getElementById(`stat-value-${i}-${j}`).value

                        });

                    }

                    data.cards.push(cardData);

                }

                localStorage.setItem(STORAGE_KEY, JSON.stringify(data));

            }


            function loadData() {

                const savedData = localStorage.getItem(STORAGE_KEY);

                if (!savedData) return;

                

                try {

                    const data = JSON.parse(savedData);

                    document.getElementById('budget').value = data.budget || '100000';

                    data.cards.forEach((cardData, i) => {

                        if (i < cardCount) {

                            document.getElementById(`level-${i}`).value = cardData.level || '1';

                            document.getElementById(`awaken-${i}`).value = cardData.awaken || '0';

                            cardData.stats.forEach((stat, j) => {

                                document.getElementById(`stat-type-${i}-${j}`).value = stat.type || 'NONE';

                                document.getElementById(`stat-value-${i}-${j}`).value = stat.value || '0';

                            });

                        }

                    });

                } catch(e) {

                    console.error("저장된 데이터를 불러오는 데 실패했습니다.", e);

                    localStorage.removeItem(STORAGE_KEY);

                }

            }


            createCardInputs();

            function runCalculation() {

                try {

                    const initialBudget = parseInt(document.getElementById('budget').value, 10);

                    let budgetLeft = initialBudget;

                    

                    const cards = [];

                    for (let i = 0; i < cardCount; i++) {

                        const stats = [];

                        for(let j=0; j<3; j++) {

                            stats.push({

                                type: document.getElementById(`stat-type-${i}-${j}`).value,

                                value: parseFloat(document.getElementById(`stat-value-${i}-${j}`).value) || 0,

                                threshold: THRESHOLDS[j]

                            });

                        }


                        cards.push({

                            id: i,

                            startLevel: parseInt(document.getElementById(`level-${i}`).value, 10) || 0,

                            plannedLevel: parseInt(document.getElementById(`level-${i}`).value, 10) || 0,

                            awaken: parseInt(document.getElementById(`awaken-${i}`).value, 10) || 0,

                            stats: stats

                        });

                    }


                    const getLevelCap = (awaken) => 150 + 50 * awaken;

                    const getBrushCost = (level) => 1500 + 20 * (level - 1);

                    const m = (s) => (s > 0) ? (s / 100) : 1;


                    const calculateStatsProduct = (cardStates) => {

                        const S = { ATK: 0, CRIT: 0, FINAL: 0, AFFIN: 0 };

                        cardStates.forEach(card => {

                            if (card.plannedLevel <= 0) return;

                            card.stats.forEach(stat => {

                                if (card.awaken >= stat.threshold && stat.type !== 'NONE') {

                                    S[stat.type] += stat.value * card.plannedLevel;

                                }

                            });

                        });

                        return m(S.ATK) * m(S.CRIT) * m(S.FINAL) * m(S.AFFIN);

                    };


                    const initialProduct = calculateStatsProduct(cards);

                    let currentProduct = initialProduct;


                    const maxSteps = 20000;

                    let steps = 0;

                    

                    while (budgetLeft > 0 && steps < maxSteps) {

                        let bestCard = -1;

                        let bestROI = -1;


                        for (let i = 0; i < cards.length; i++) {

                            const card = cards[i];

                            const cap = getLevelCap(card.awaken);


                            if (card.plannedLevel < 1 || card.plannedLevel >= cap) continue;

                            const cost = getBrushCost(card.plannedLevel);

                            if (cost > budgetLeft) continue;

                            

                            const nextStates = JSON.parse(JSON.stringify(cards));

                            nextStates[i].plannedLevel++;

                            

                            const nextProduct = calculateStatsProduct(nextStates);

                            const roi = (nextProduct - currentProduct) / cost;

                            

                            if (roi > bestROI) {

                                bestROI = roi;

                                bestCard = i;

                            }

                        }


                        if (bestCard === -1 || bestROI <= 0) break;


                        const cost = getBrushCost(cards[bestCard].plannedLevel);

                        budgetLeft -= cost;

                        cards[bestCard].plannedLevel++;

                        currentProduct = calculateStatsProduct(cards);

                        steps++;

                    }


                    displayResults(cards, initialBudget, budgetLeft, initialProduct, currentProduct);


                } catch (e) {

                    console.error(e);

                    statusDiv.textContent = `오류가 발생했습니다: ${e.message}`;

                    statusDiv.classList.add('bg-red-100', 'text-red-700');

                }

            }


            function displayResults(cards, initialBudget, budgetLeft, initialProduct, finalProduct) {

                let planHtml = '<ul>';

                let changes = 0;

                cards.forEach(card => {

                    if (card.plannedLevel > card.startLevel) {

                        planHtml += `<li class="py-1"><strong>카드 ${card.id + 1}:</strong> ${card.startLevel} → <span class="font-bold text-green-600">${card.plannedLevel}</span> 레벨</li>`;

                        changes++;

                    }

                });

                if (changes === 0) {

                    planHtml = '<p>레벨업 할 카드가 없거나 예산이 부족합니다.</p>';

                }

                planHtml += '</ul>';


                const totalGain = initialProduct > 0 ? ((finalProduct / initialProduct - 1) * 100) : 0;


                planOutput.innerHTML = planHtml;

                document.getElementById('used-brush').textContent = (initialBudget - budgetLeft).toLocaleString();

                document.getElementById('remaining-brush').textContent = budgetLeft.toLocaleString();

                document.getElementById('total-gain').textContent = `${totalGain.toFixed(4)}%`;


                if (changes > 0) {

                    statusDiv.textContent = '계산이 완료되었습니다.';

                    statusDiv.classList.remove('bg-red-100', 'text-red-700');

                } else {

                    statusDiv.textContent = '더 이상 효율적으로 레벨업할 카드가 없습니다.';

                }

            }


        });
